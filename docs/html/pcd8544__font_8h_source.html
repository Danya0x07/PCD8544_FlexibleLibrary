222 static const uint8_t *_get_character_bitmap(char c)
223 {
224  const uint8_t *bitmap = (const uint8_t *)&ascii_charset[c - 0x20];
225 
226  if (c >= 0x80) { // non-ASCII character
227  switch (current_language)
228  {
229  case PCD8544_LANG_RU:
230  if (c < 0xE0)
231  c -= 0x80;
232  else
233  c -= 0xB0;
234  bitmap = (const uint8_t *)&russian_charset[c];
235  // Other languages - other font tables and manipulations with indicies.
236  }
237  }
238 
239  return bitmap;
240 }
